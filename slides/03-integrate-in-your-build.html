3. Integrate in your build

// Grunt
module.exports = function (grunt) {
  grunt.initConfig({
    ts: {
      build: {
        tsconfig: true
      }
  }
}
// Gulp
var ts = require('gulp-typescript');
 
gulp.task('default', function () {
    return gulp.src('src/**/*.ts')
      .pipe(ts())
      .pipe(gulp.dest('dist'));

});

// Package.json
{
    "scripts": {
        "start": "tsc"
    }
}
